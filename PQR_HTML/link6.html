ver parentesco para segmentar usuario con enfermedades y padecimientos cronicos



SELECT datospersonales.nombre, antecedentes.nombrePadecimiento, antecedentes.estadoCronico FROM datospersonales INNER JOIN antecedentes ON antecedentes.numeroAntecedente = datospersonales.numeroAntecedente WHERE datospersonales.curp IN ( SELECT curp_usuario FROM familiares WHERE parentesco = 'amigo' ) ORDER BY datospersonales.nombre ASC, antecedentes.nombrePadecimiento ASC;
